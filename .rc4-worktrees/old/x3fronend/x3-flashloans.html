HOW IT
WORKS.

Four phases. Cross-chain atomic execution. No exceptions.

🔒
Lock HTLCs

Secret S and hash H=sha256(S) are generated. Funds lock into HTLC contracts on both Solana (T_S) and Ethereum (T_E) with staggered timelocks.

// PHASE 01
1
⚡
Fire Flashloans

The planner picks the optimal provider per chain. Kamino/MarginFi for SVM, Balancer/Aave for EVM. Borrow → Swap → Repay. One TX each.

// PHASE 02
2
🔑
Reveal Secret

Both legs succeed? Initiator calls claimHTLC(S) on Solana. Secret emits on-chain. Relayer picks it up and fires claimHTLC on Ethereum within milliseconds.

// PHASE 03
3
💰
Collect Profit

Both HTLCs release funds to initiator. Flash pools return to full liquidity. If anything failed — no secret, both chains refund after timeouts. Zero loss.

// X3-lang HTLC contract — x3vm WASM runtime
contract AtomicHTLC {

  struct Htlc {
    sender:    Address,
    recipient: Address,
    hash_lock: [u8; 32],
    timelock:  u64,
    asset:     AssetId,
    amount:    u128,
    claimed:   bool,
  }

  pub fn create_htlc(
    recipient: Address, hash_lock: [u8; 32],
    timelock: u64,   asset: AssetId, amount: u128
  ) -> HtlcId {
    assert!(amount > 0, "NonzeroAmount");
    assert!(timelock > block_time(), "FutureTimelock");
    // lock asset from caller, store HTLC
    emit HtlcCreated { id, recipient, hash_lock, timelock };
    id
  }

  pub fn claim_htlc(id: HtlcId, secret: [u8; 32]) {
    let h = sha256(secret);
    assert!(h == htlc.hash_lock, "InvalidSecret");
    assert!(block_time() < htlc.timelock, "TimelockExpired");
    htlc.claimed = true;
    transfer(htlc.recipient, htlc.asset, htlc.amount);
    emit HtlcClaimed { id, secret };  // relayer watches this
  }

  pub fn refund_htlc(id: HtlcId) {
    assert!(block_time() >= htlc.timelock, "TooEarly");
    transfer(htlc.sender, htlc.asset, htlc.amount);
    emit HtlcRefunded { id };
  }
}
// AtlasHTLC.sol — EVM atomic HTLC with flashloan adapter
interface IFlashLoanReceiver {
  function executeOperation(
    address[] calldata assets, uint256[] calldata amounts,
    uint256[] calldata premiums, address initiator, bytes calldata params
  ) external returns (bool);
}

contract AtlasHTLC is IFlashLoanReceiver, ReentrancyGuard {

  function createHTLC(
    address recipient, bytes32 hashLock,
    uint256 timeLock, address token, uint256 amount
  ) external returns (bytes32 id) {
    // transfer tokens in, store HTLC struct
    emit HTLCCreated(id, recipient, hashLock, timeLock);
  }

  function executeOperation(/* Aave callback */) external returns (bool) {
    // decode params: swap instructions
    IUniswapV3Router(router).exactInputSingle(swapParams);
    // approve Aave pool to pull back principal + premium
    IERC20(asset).approve(aavePool, amount + premium);
    return true;
  }

  function claimHTLC(bytes32 id, bytes32 secret)
    external nonReentrant {
    require(keccak256(abi.encode(secret)) == htlc.hashLock);
    require(block.timestamp < htlc.timeLock);
    IERC20(htlc.token).transfer(htlc.recipient, htlc.amount);
    emit HTLCClaimed(id, secret);
  }
}

#[program]
pub mod x3_htlc {
  pub fn create_htlc(ctx: Context, params: HtlcParams) -> Result<()> {
    let htlc = &mut ctx.accounts.htlc;
    require!(params.amount > 0, HtlcError::NonzeroAmount);
    require!(params.timelock > Clock::get()?.unix_timestamp, ...);
    // CPI: transfer USDC to HTLC PDA
    token::transfer(cpi_ctx, params.amount)?;
    emit!(HtlcCreated { ... });
    Ok(())
  }

  pub fn kamino_flash_arb(ctx: Context, amount: u64) -> Result<()> {
    // Kamino borrow instruction (same TX)
    let borrow_ix = kamino::flash_borrow_ix(&ctx.accounts, amount);
    invoke(&borrow_ix, accounts)?;
    // DEX swap via Jupiter aggregator
    jupiter::route_swap(USDC, SOL, amount, min_out)?;
    // Kamino repay (same TX — atomic)
    let repay_ix = kamino::flash_repay_ix(&ctx.accounts, amount);
    invoke(&repay_ix, accounts)?;
    Ok(())
  }

  pub fn claim_htlc(ctx: Context, secret: [u8; 32]) -> Result<()> {
    let h = hash::hashv(&[&secret]);
    require!(h.to_bytes() == htlc.hash_lock, InvalidSecret);
    token::transfer(cpi_ctx, htlc.amount)?;
    emit!(HtlcClaimed { secret }); // relayer observes, claims EVM
    Ok(())
  }
}
